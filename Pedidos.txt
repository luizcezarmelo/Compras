        async function saveFileHandle(handle) {

        async function getSavedFileHandle() {
            try {
                const db = await openDB();
                const transaction = db.transaction(['fileHandles'], 'readonly');
                const store = transaction.objectStore('fileHandles');
                const handle = await store.get(FILE_HANDLE_KEY);
                return handle;
            } catch (error) {
                console.warn('N√£o foi poss√≠vel recuperar a refer√™ncia do arquivo:', error);
                return null;
            }
        }

        async function verifyPermission(handle) {
            try {
                if (handle.queryPermission) {
                    const options = { mode: 'readwrite' };
                    if ((await handle.queryPermission(options)) === 'granted') {
                        return true;
                    }
                    if ((await handle.requestPermission(options)) === 'granted') {
                        return true;
                    }
                } else {
                    const file = await handle.getFile();
                    if (file) return true;
                }
                return false;
            } catch (error) {
                console.warn('Erro ao verificar permiss√£o:', error);
                return false;
            }
        }

        function openDB() {
            return new Promise((resolve, reject) => {
                const request = indexedDB.open('PedidosDB', 1);
                request.onerror = () => reject(request.error);
                request.onsuccess = () => resolve(request.result);
                request.onupgradeneeded = (event) => {
                    const db = event.target.result;
                    if (!db.objectStoreNames.contains('fileHandles')) {
                        db.createObjectStore('fileHandles');
                    }
                };
            });
        }

        function makeColumnsResizable() {
            const headers = document.querySelectorAll('th');
            headers.forEach((header) => {
                const oldHandle = header.querySelector('.resize-handle');
                if (oldHandle) oldHandle.remove();

                const key = header.getAttribute('data-key');
                if (key && columnSizes[key]) {
                    header.style.minWidth = columnSizes[key] + 'px';
                    header.style.maxWidth = 'none';
                }

                const resizeHandle = document.createElement('div');
                resizeHandle.className = 'resize-handle';
                header.style.position = 'relative';
                header.appendChild(resizeHandle);

                let startX, startWidth;

                resizeHandle.addEventListener('mousedown', initResize);

                function initResize(e) {
                    startX = e.clientX;
                    startWidth = header.offsetWidth;
                    document.documentElement.style.cursor = 'col-resize';
                    document.body.style.userSelect = 'none';
                    document.addEventListener('mousemove', doResize);
                    document.addEventListener('mouseup', stopResize);
                }

                function doResize(e) {
                    const width = startWidth + (e.clientX - startX);
                    if (width > 30) {
                        header.style.minWidth = width + 'px';
                        header.style.maxWidth = 'none';

                        const key = header.getAttribute('data-key');
                        if (key) {
                            columnSizes[key] = width;
                            const cells = document.querySelectorAll(`td[data-key="${key}"]`);
                            cells.forEach(cell => {
                                cell.style.minWidth = width + 'px';
                                cell.style.maxWidth = 'none';
                            });
                        }
                    }
                }

                function stopResize() {
                    document.documentElement.style.cursor = '';
                    document.body.style.userSelect = '';
                    document.removeEventListener('mousemove', doResize);
                    document.removeEventListener('mouseup', stopResize);
                    if (isEdit) {
                        debouncedSave();
                    }
                }
            });
        }

        function getTodayDate() {
            const d = new Date();
            const day = String(d.getDate()).padStart(2, '0');
            const month = String(d.getMonth() + 1).padStart(2, '0');
            const year = d.getFullYear();
            return `${day}/${month}/${year}`;
        }

        function getFilenameTimestamp() {
            const d = new Date();
            const dStr = String(d.getDate()).padStart(2, '0') + String(d.getMonth() + 1).padStart(2, '0') + d.getFullYear();
            const tStr = String(d.getHours()).padStart(2, '0') + String(d.getMinutes()).padStart(2, '0');
            return `${dStr}_${tStr}`;
        }

        function maskPedidoCell(cellElement, id) {
            if (!isEdit) return;

            // Salva posi√ß√£o do cursor
            const selection = window.getSelection();
            const range = selection.getRangeAt(0);
            const cursorOffset = range.startOffset;

            // Aplica m√°scara
            let text = cellElement.innerText;
            let v = text.replace(/\D/g, '');
            
            if (v.length > 0 && !v.startsWith('4500')) {
                v = '4500' + v.replace(/^4500/, '');
            }
            
            if (v.length < 4) {
                v = '4500';
            }
            
            if (v.length > 10) {
                v = v.substring(0, 10);
            }

            // Atualiza c√©lula se mudou
            if (cellElement.innerText !== v) {
                cellElement.innerText = v;

                // Restaura posi√ß√£o do cursor
                try {
                    const textNode = cellElement.firstChild;
                    if (textNode) {
                        const newRange = document.createRange();
                        const newOffset = Math.min(cursorOffset, v.length);
                        newRange.setStart(textNode, newOffset);
                        newRange.collapse(true);
                        selection.removeAllRanges();
                        selection.addRange(newRange);
                    }
                } catch (e) {
                    // Ignora erros de restaura√ß√£o de cursor
                }
            }
        }

        function maskDate(el) {
            let v = el.value.replace(/[^0-9]/g, '');
            let day = v.substring(0, 2);
            let month = v.substring(2, 4);

            if (day.length === 2) {
                day = parseInt(day);
                if (day > 31) day = 31;
                day = String(day).padStart(2, '0');
            }

            if (month.length === 2) {
                month = parseInt(month);
                if (month > 12) month = 12;
                month = String(month).padStart(2, '0');
            }

            if (v.length > 2) el.value = day + '/' + month;
            else el.value = day;
        }

        function maskMoney(el) {
            let v = el.value.replace(/\D/g, '');
            if (v.length === 0) {
                el.value = '';
                return;
            }
            v = v.padStart(3, '0');
            v = v.replace(/(\d)(\d{2})$/, '$1,$2');
            v = v.replace(/(\d+)(\d{3},\d{2})$/, '$1.$2');
            v = v.replace(/(\d+)(\d{3}\.\d{3},\d{2})$/, '$1.$2');
            v = v.replace(/^0+(?=\d)/, '');
            el.value = (v.length > 0) ? 'R$ ' + v : '';
        }

        function formatCellDate(text) {
            let v = text.replace(/[^0-9]/g, '');
            let day = v.substring(0, 2);
            let month = v.substring(2, 4);

            if (day.length === 2) {
                day = parseInt(day);
                if (day > 31) day = 31;
                day = String(day).padStart(2, '0');
            }

            if (month.length === 2) {
                month = parseInt(month);
                if (month > 12) month = 12;
                month = String(month).padStart(2, '0');
            }

            if (v.length >= 3) return day + '/' + month;
            else if (v.length > 0) return day;
            return '';
        }

        function navInput(e, nextId, isLast = false) {
            const isEnter = e.key.toUpperCase() === 'ENTER';
            const isTabForward = e.key.toUpperCase() === 'TAB' && !e.shiftKey;

            if (isEnter || isTabForward) {
                e.preventDefault();

        }

        function navTable(event, currentCell) {
            const isEnter = event.key.toUpperCase() === 'ENTER';
            const isTabForward = event.key.toUpperCase() === 'TAB' && !event.shiftKey;

            if (!isEnter && !isTabForward) return;
            
            event.preventDefault();

            const row = currentCell.parentElement;
            const cells = Array.from(row.querySelectorAll('td[contenteditable="true"]'));
            const currentIndex = cells.indexOf(currentCell);

            if (currentIndex < cells.length - 1) {
                cells[currentIndex + 1].focus();
            } else {
                const nextRow = row.nextElementSibling;
                if (nextRow) {
                    const nextCells = nextRow.querySelectorAll('td[contenteditable="true"]');
                    if (nextCells.length > 0) {
                        nextCells[0].focus();
                    }
                }
            }
        }

        function parseDeliveryDate(dateStr) {
            if (!dateStr || dateStr.length !== 5) return null;
            const parts = dateStr.split('/');
            if (parts.length !== 2) return null;
            let day = parseInt(parts[0], 10);
            let month = parseInt(parts[1], 10) - 1;
            const today = new Date();
            today.setHours(0, 0, 0, 0);
            const currentYear = today.getFullYear();
            let date = new Date(currentYear, month, day);
            if (date < today) {
                date.setFullYear(currentYear + 1);
            }
            if (date.getDate() !== day || date.getMonth() !== month) return null;
            return date;
        }

        function downloadData(data, prefix) {
            const filename = `${prefix}_${getFilenameTimestamp()}.json`;
            const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });

        async function doRestore() {
            if (!isEdit) {
                return;
            }
            if (!confirm('ATEN√á√ÉO: Isso ir√° substituir os dados atuais.\n\nContinuar?')) return;

            const input = document.createElement('input');
            input.type = 'file';
            input.accept = '.json';
            input.onchange = async (e) => {
                const file = e.target.files[0];
                if (!file) return;

                const reader = new FileReader();
                reader.onload = async function (e) {
                    try {
                        const restoredData = JSON.parse(e.target.result);
                        db = restoredData.db_pedidos || db;
                        trash = restoredData.db_lixeira || trash;
                        if (restoredData.columnWidths) {
                            columnSizes = restoredData.columnWidths;
                            applyColumnSizes();
                        }
                        db.sort((a, b) => a.id - b.id);
                        await saveToFile();
                        renderMain();
                    } catch (err) {
                        alert('Erro ao ler o arquivo JSON de backup.');
                    }
                };
                reader.readAsText(file);
            };
            input.click();
        }

        function isBusinessDay(date) {
            const day = date.getDay();
            return day !== 0 && day !== 6;
        }

        function addBusinessDays(startDate, daysToAdd) {
            let currentDate = new Date(startDate);
            let addedDays = 0;

            while (addedDays < daysToAdd) {
                currentDate.setDate(currentDate.getDate() + 1);
                if (isBusinessDay(currentDate)) {
                    addedDays++;
                }
            }

            return currentDate;
        }

        function calculateDate() {
            const daysInput = document.getElementById('calc-days');
            const resultSpan = document.getElementById('calc-result');
            const isUteis = document.querySelector('input[name="day-type"]:checked').value === 'uteis';
            const days = parseInt(daysInput.value, 10);

            if (!days || days <= 0) {
                resultSpan.innerText = '--/--';
                return;
            }

            const today = new Date();
            today.setHours(0, 0, 0, 0);

            let targetDate;
            if (isUteis) {
                targetDate = addBusinessDays(today, days);
            } else {
                targetDate = new Date(today);
                targetDate.setDate(today.getDate() + days);
            }

            const day = String(targetDate.getDate()).padStart(2, '0');
            const month = String(targetDate.getMonth() + 1).padStart(2, '0');
            resultSpan.innerText = `${day}/${month}`;
        }
